  pinMode(A2,OUTPUT);
  pinMode(A3,OUTPUT);
  pinMode(A4,OUTPUT);
  Serial.begin(9600);
}

void loop() {
  analogWrite(osol,255); // 255 yerine bluetooth dan gelen
  analogWrite(osag,255); // verileri yerleþtiricezzz
  analogWrite(asol,255); // galiba Serial.read(A5);
  analogWrite(asag,255); // ama analog giriþlerimiz bitti wtf
}
